## TRASH-SAVES


from pydub import AudioSegment


class SplitWavAudio:
    def __init__(self, pasta, ficheiro):
        self.pasta = pasta
        self.ficheiro = ficheiro
        self.filepath = pasta + '/' + ficheiro

        self.audio = AudioSegment.from_wav(self.filepath)

    def divisao(self, from_min, to_min, split_filename):
        t1 = from_min * 60 * 1000
        t2 = to_min * 60 * 1000
        split_audio = self.audio[t1:t2]
        split_audio.export(self.pasta + '/' + split_filename, format="wav")


pasta = 'Som'
ficheiro = 'som1.wav'
split_wav = SplitWavAudio(pasta, ficheiro)
split_wav.divisao(1, 2, 'som2.wav')
